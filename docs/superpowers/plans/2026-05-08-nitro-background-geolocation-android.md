# Nitro Background Geolocation — Android Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement a React Native Nitro Module that wraps the cordova-background-geolocation-plugin's Android native code, exposing it via a typed JSI bridge with an idiomatic React Native API.

**Architecture:** The Cordova plugin's `common/` Java source files (~65 files) are copied into the Nitro module's Android source set. A new Kotlin `NitroBackgroundGeolocation` class implements `HybridNitroBackgroundGeolocationSpec` (generated by Nitrogen from the `.nitro.ts` spec) and delegates all calls to the existing `BackgroundGeolocationFacade` Java class, implementing its `PluginDelegate` interface to forward events to JS callbacks. A headless task service extends React Native's `HeadlessJsTaskService` to fire user-registered tasks when the app is killed.

**Tech Stack:** Nitro Modules (JSI bridge), Nitrogen (code generator), Kotlin (bridge layer), Java (copied native code), React Native New Architecture, TypeScript

This class converts between Nitrogen-generated structs and the Java `Config` class. The exact struct class names depend on what Nitrogen generates — check `nitrogen/generated/android/kotlin/` after running `yarn nitrogen` in Task 1.

```kotlin
package com.margelo.nitro.nitrobackgroundgeolocation

import com.marianhello.bgloc.Config
import com.marianhello.bgloc.data.BackgroundActivity
import com.marianhello.bgloc.data.BackgroundLocation
import com.marianhello.logging.LogEntry as NativeLogEntry

object ConfigMapper {

    fun toNativeConfig(options: ConfigureOptions): Config {
        val config = Config()
        options.headlessTaskName // stored separately, not in Config

        options.locationProvider?.let { config.setLocationProvider(it.ordinal) }
        options.desiredAccuracy?.let { config.setDesiredAccuracy(it.value.toInt()) }
        options.stationaryRadius?.let { config.setStationaryRadius(it.toFloat()) }
        options.debug?.let { config.setDebugging(it) }
        options.distanceFilter?.let { config.setDistanceFilter(it.toInt()) }
        options.stopOnTerminate?.let { config.setStopOnTerminate(it) }
        options.startOnBoot?.let { config.setStartOnBoot(it) }
        options.interval?.let { config.setInterval(it.toInt()) }
        options.fastestInterval?.let { config.setFastestInterval(it.toInt()) }
        options.activitiesInterval?.let { config.setActivitiesInterval(it.toInt()) }
        options.stopOnStillActivity?.let { config.setStopOnStillActivity(it) }
        options.notificationsEnabled?.let { config.setNotificationsEnabled(it) }
        options.startForeground?.let { config.setStartForeground(it) }
        options.notificationTitle?.let { config.setNotificationTitle(it) }
        options.notificationText?.let { config.setNotificationText(it) }
        options.notificationIconColor?.let { config.setNotificationIconColor(it) }
        options.notificationIconLarge?.let { config.setLargeNotificationIcon(it) }
        options.notificationIconSmall?.let { config.setSmallNotificationIcon(it) }
        options.url?.let { config.setUrl(it) }
        options.syncUrl?.let { config.setSyncUrl(it) }
        options.syncThreshold?.let { config.setSyncThreshold(it.toInt()) }
        options.httpHeaders?.let {
            val hashMap = HashMap<String, String>()
            it.forEach { (k, v) -> hashMap[k] = v }
            config.setHttpHeaders(hashMap)
        }
        options.maxLocations?.let { config.setMaxLocations(it.toInt()) }
        options.postTemplate?.let { config.setTemplate(/* convert to LocationTemplate */ ) }

        return config
    }

    fun toJsConfig(config: Config): ConfigureOptions {
        return ConfigureOptions(
            headlessTaskName = "", // not stored in native Config
            locationProvider = config.locationProvider?.let { LocationProvider.entries[it] },
            desiredAccuracy = config.desiredAccuracy?.let { acc ->
                LocationAccuracy.entries.find { it.value.toInt() == acc }
            },
            stationaryRadius = config.stationaryRadius?.toDouble(),
            debug = config.isDebugging,
            distanceFilter = config.distanceFilter?.toDouble(),
            stopOnTerminate = config.stopOnTerminate,
            startOnBoot = config.startOnBoot,
            interval = config.interval?.toDouble(),
            fastestInterval = config.fastestInterval?.toDouble(),
            activitiesInterval = config.activitiesInterval?.toDouble(),
            stopOnStillActivity = config.stopOnStillActivity,
            notificationsEnabled = config.notificationsEnabled,
            startForeground = config.startForeground,
            notificationTitle = config.notificationTitle,
```

```kotlin
package com.margelo.nitro.nitrobackgroundgeolocation

import android.content.Intent
import android.os.Bundle
import com.facebook.react.HeadlessJsTaskService
import com.facebook.react.bridge.Arguments
import com.facebook.react.jstasks.HeadlessJsTaskConfig

class HeadlessTaskService : HeadlessJsTaskService() {

    override fun getTaskConfig(intent: Intent): HeadlessJsTaskConfig? {
        val extras = intent.extras ?: return null
        val taskName = extras.getString("taskName") ?: return null
        val eventData = Arguments.createMap().apply {
            putString("name", extras.getString("eventName", ""))
            putString("params", extras.getString("eventParams", ""))
        }
        return HeadlessJsTaskConfig(
            taskName,
            eventData,
            5000L,
            true
        )
    }

    companion object {
        fun createIntent(
            service: android.content.Context,
            taskName: String,
            eventName: String,
            eventParams: String
        ): Intent {
            return Intent(service, HeadlessTaskService::class.java).apply {
                putExtra("taskName", taskName)
                putExtra("eventName", eventName)
                putExtra("eventParams", eventParams)
            }
        }
    }
}
```

```kotlin
package com.margelo.nitro.nitrobackgroundgeolocation

import android.content.Context
import com.facebook.proguard.annotations.DoNotStrip
import com.facebook.react.bridge.ReactApplicationContext
import com.facebook.react.modules.core.DeviceEventManagerModule
import com.marianhello.bgloc.BackgroundGeolocationFacade

import com.marianhello.bgloc.data.BackgroundActivity
import com.marianhello.bgloc.data.BackgroundLocation
import com.margelo.nitro.core.Promise
import com.marianhello.logging.LoggerManager

@DoNotStrip
class NitroBackgroundGeolocation(
    private val context: Context
) : HybridNitroBackgroundGeolocationSpec(), PluginDelegate {

    private val logger = LoggerManager.getLogger(NitroBackgroundGeolocation::class.java)
    private val facade = BackgroundGeolocationFacade(context, this)
    private var headlessTaskName: String = ""

    // Callback storage
    private val locationCallbacks = mutableListOf<(Location) -> Unit>()
    private val stationaryCallbacks = mutableListOf<(StationaryLocation) -> Unit>()
    private val activityCallbacks = mutableListOf<(Activity) -> Unit>()
    private val startCallbacks = mutableListOf<() -> Unit>()
    private val stopCallbacks = mutableListOf<() -> Unit>()
    private val errorCallbacks = mutableListOf<(BackgroundGeolocationError) -> Unit>()
    private val authorizationCallbacks = mutableListOf<(AuthorizationStatus) -> Unit>()
    private val foregroundCallbacks = mutableListOf<() -> Unit>()
    private val backgroundCallbacks = mutableListOf<() -> Unit>()
    private val abortRequestedCallbacks = mutableListOf<() -> Unit>()
    private val httpAuthorizationCallbacks = mutableListOf<() -> Unit>()

    // --- Lifecycle ---

    override fun configure(options: ConfigureOptions): Promise<Unit> {
        return Promise.async {
            headlessTaskName = options.headlessTaskName
            val config = ConfigMapper.toNativeConfig(options)
            facade.configure(config)
        }
    }

    override fun start(): Promise<Unit> {
        return Promise.async {
            facade.start()
        }
    }

    override fun stop(): Promise<Unit> {
        return Promise.async {
            facade.stop()
        }
    }

    // --- Location ---

    override fun getCurrentLocation(options: LocationOptions): Promise<Location> {
        return Promise.async {
            val timeout = options.timeout?.toInt() ?: 60000
            val maximumAge = options.maximumAge?.toLong() ?: 0L
            val enableHighAccuracy = options.enableHighAccuracy ?: false
            val location = facade.getCurrentLocation(timeout, maximumAge, enableHighAccuracy)
            ConfigMapper.toJsLocation(location)
        }
    }

    override fun getStationaryLocation(): Promise<StationaryLocation?> {
        return Promise.async {
            val location = facade.stationaryLocation
            location?.let { ConfigMapper.toJsStationaryLocation(it) }
        }
    }

    // --- Status ---

    override fun checkStatus(): Promise<ServiceStatus> {
        return Promise.async {
            ServiceStatus(
                isRunning = facade.isRunning,
                locationServicesEnabled = facade.locationServicesEnabled(),
                authorization = when (facade.authorizationStatus) {
                    BackgroundGeolocationFacade.AUTHORIZATION_AUTHORIZED ->
                        AuthorizationStatus.AUTHORIZED
                    else -> AuthorizationStatus.NOT_AUTHORIZED
                }
            )
        }
    }

    override fun getConfig(): Promise<ConfigureOptions> {
        return Promise.async {
            val config = facade.config
            ConfigMapper.toJsConfig(config).let { jsConfig ->
                // Patch headlessTaskName back in since it's not in native Config
                ConfigureOptions(
                    headlessTaskName = headlessTaskName,
                    locationProvider = jsConfig.locationProvider,
                    desiredAccuracy = jsConfig.desiredAccuracy,
                    stationaryRadius = jsConfig.stationaryRadius,
                    debug = jsConfig.debug,
                    distanceFilter = jsConfig.distanceFilter,
                    stopOnTerminate = jsConfig.stopOnTerminate,
                    startOnBoot = jsConfig.startOnBoot,
                    interval = jsConfig.interval,
                    fastestInterval = jsConfig.fastestInterval,
                    activitiesInterval = jsConfig.activitiesInterval,
                    stopOnStillActivity = jsConfig.stopOnStillActivity,
                    notificationsEnabled = jsConfig.notificationsEnabled,
                    startForeground = jsConfig.startForeground,
                    notificationTitle = jsConfig.notificationTitle,
                    notificationText = jsConfig.notificationText,
                    notificationIconColor = jsConfig.notificationIconColor,
                    notificationIconLarge = jsConfig.notificationIconLarge,
                    notificationIconSmall = jsConfig.notificationIconSmall,
                    activityType = jsConfig.activityType,
                    pauseLocationUpdates = jsConfig.pauseLocationUpdates,
                    saveBatteryOnBackground = jsConfig.saveBatteryOnBackground,
                    url = jsConfig.url,
                    syncUrl = jsConfig.syncUrl,
                    syncThreshold = jsConfig.syncThreshold,
                    httpHeaders = jsConfig.httpHeaders,
                    maxLocations = jsConfig.maxLocations,
                    postTemplate = jsConfig.postTemplate,
                )
            }
        }
    }

    // --- Storage ---

    override fun getLocations(): Promise<Array<Location>> {
        return Promise.async {
            facade.locations.map { ConfigMapper.toJsLocation(it) }.toTypedArray()
        }
    }

    override fun getValidLocations(): Promise<Array<Location>> {
        return Promise.async {
            facade.validLocations.map { ConfigMapper.toJsLocation(it) }.toTypedArray()
        }
    }

    override fun getValidLocationsAndDelete(): Promise<Array<Location>> {
        return Promise.async {
            facade.validLocationsAndDelete.map { ConfigMapper.toJsLocation(it) }.toTypedArray()
        }
    }

    override fun deleteLocation(locationId: Double): Promise<Unit> {
        return Promise.async {
            facade.deleteLocation(locationId.toLong())
        }
    }

    override fun deleteAllLocations(): Promise<Unit> {
        return Promise.async {
            facade.deleteAllLocations()
        }
    }

    // --- Sync ---

    override fun forceSync(): Promise<Unit> {
        return Promise.async {
            facade.forceSync()
        }
    }

    // --- Logging ---

    override fun getLogEntries(
        limit: Double,
        fromId: Double,
        minLevel: LogLevel
    ): Promise<Array<LogEntry>> {
        return Promise.async {
            val levelStr = when (minLevel) {
                LogLevel.TRACE -> "TRACE"
                LogLevel.DEBUG -> "DEBUG"
                LogLevel.INFO -> "INFO"
                LogLevel.WARN -> "WARN"
                LogLevel.ERROR -> "ERROR"

    override fun onLocation(callback: (Location) -> Unit): () -> Unit {
        locationCallbacks.add(callback)
        return { locationCallbacks.remove(callback) }

        activityCallbacks.add(callback)
        return { activityCallbacks.remove(callback) }

    override fun onStart(callback: () -> Unit): () -> Unit {
        startCallbacks.add(callback)
        return { startCallbacks.remove(callback) }
    }

    override fun onStop(callback: () -> Unit): () -> Unit {
        stopCallbacks.add(callback)
        return { stopCallbacks.remove(callback) }
    }

    override fun onError(callback: (BackgroundGeolocationError) -> Unit): () -> Unit {
        errorCallbacks.add(callback)
        return { errorCallbacks.remove(callback) }
    }

    override fun onAuthorization(callback: (AuthorizationStatus) -> Unit): () -> Unit {
        authorizationCallbacks.add(callback)
        return { authorizationCallbacks.remove(callback) }
    }

    override fun onForeground(callback: () -> Unit): () -> Unit {
        foregroundCallbacks.add(callback)
        return { foregroundCallbacks.remove(callback) }
    }

    override fun onBackground(callback: () -> Unit): () -> Unit {
        backgroundCallbacks.add(callback)
        return { backgroundCallbacks.remove(callback) }
    }

    override fun onAbortRequested(callback: () -> Unit): () -> Unit {
        abortRequestedCallbacks.add(callback)
        return { abortRequestedCallbacks.remove(callback) }
    }

    override fun onHttpAuthorization(callback: () -> Unit): () -> Unit {
        httpAuthorizationCallbacks.add(callback)
        return { httpAuthorizationCallbacks.remove(callback) }
    }

    override fun removeAllListeners() {
        locationCallbacks.clear()
        stationaryCallbacks.clear()
        activityCallbacks.clear()
        startCallbacks.clear()
        stopCallbacks.clear()
        errorCallbacks.clear()
        authorizationCallbacks.clear()
        foregroundCallbacks.clear()
        backgroundCallbacks.clear()
        abortRequestedCallbacks.clear()
        httpAuthorizationCallbacks.clear()
    }

    // --- PluginDelegate implementation ---

    override fun onLocationChanged(location: BackgroundLocation) {
        val jsLocation = ConfigMapper.toJsLocation(location)
        locationCallbacks.forEach { it(jsLocation) }
        fireHeadlessTask("location", location)
    }

    override fun onStationaryChanged(location: BackgroundLocation) {
        val jsLocation = ConfigMapper.toJsStationaryLocation(location)
        stationaryCallbacks.forEach { it(jsLocation) }
        fireHeadlessTask("stationary", location)
    }

    override fun onActivityChanged(activity: BackgroundActivity) {
        val jsActivity = ConfigMapper.toJsActivity(activity)
        activityCallbacks.forEach { it(jsActivity) }
    }

    override fun onServiceStatusChanged(status: Int) {
        if (status == BackgroundGeolocationFacade.SERVICE_STARTED) {
            startCallbacks.forEach { it() }
        } else {
            stopCallbacks.forEach { it() }
        }
    }

    override fun onAuthorizationChanged(authStatus: Int) {
        val status = when (authStatus) {
            BackgroundGeolocationFacade.AUTHORIZATION_AUTHORIZED ->
                AuthorizationStatus.AUTHORIZED
            else -> AuthorizationStatus.NOT_AUTHORIZED
        }
        authorizationCallbacks.forEach { it(status) }
    }

    override fun onAbortRequested() {
        abortRequestedCallbacks.forEach { it() }
    }

    override fun onHttpAuthorization() {
        httpAuthorizationCallbacks.forEach { it() }
    }

    override fun onError(error: PluginException) {
        val jsError = BackgroundGeolocationError(
            code = error.code.toDouble(),
            message = error.message ?: "Unknown error"
        )
        errorCallbacks.forEach { it(jsError) }
    }

    // --- Headless task helper ---

    private fun fireHeadlessTask(eventName: String, location: BackgroundLocation) {
        if (headlessTaskName.isNotEmpty()) {
            try {
                val intent = HeadlessTaskService.createIntent(
                    context,
                    headlessTaskName,
                    eventName,
                    location.toJSONObject().toString()
                )
                context.startService(intent)
            } catch (e: Exception) {
                logger.error("Failed to start headless task: {}", e.message)
            }
        }
    }
}
```
